```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        num_idx_map = {}
        for idx in range(len(nums)):
            find_number = target - nums[idx]
            if find_number in num_idx_map:
                return [idx, num_idx_map[find_number]]
            num_idx_map[nums[idx]] = idx
        return [-1, -1]
```

```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        num_to_index = {}
        for index, num in enumerate(nums):
            find_num = target - num
            if find_num in num_to_index:
                find_index = num_to_index[find_num]
                return [find_index, index]
            num_to_index[num] = index
        raise Exception("unreachable")
```
