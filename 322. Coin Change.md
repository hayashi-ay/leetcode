Nの金額をを支払うのに必要な最小のコインの数をCoin(N)とするとCoin(N)はCoin(N-1)までの支払の数に影響を受けるのでDPの問題として解ける。

1st

結構さくっと解けた。
```python
class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        num_need_coins = [inf] * (amount + 1)
        num_need_coins[0] = 0

        for current_amount in range(1, amount + 1):
            for coin in coins:
                prev_amount = current_amount - coin
                if prev_amount < 0:
                    continue
                num_need_coins[current_amount] = min(
                    num_need_coins[current_amount],
                    num_need_coins[prev_amount] + 1
                )
        
        if num_need_coins[amount] == inf:
            return -1
        return num_need_coins[amount]
```

class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        num_needed_coins = [inf] * (amount + 1)
        num_needed_coins[0] = 0
        for current_amount in range(1, amount + 1):
            for coin in coins:
                if current_amount >= coin:
                    num_needed_coins[current_amount] = min(
                        num_needed_coins[current_amount],
                        num_needed_coins[current_amount - coin] + 1
                        )
        if num_needed_coins[amount] == inf:
            return -1
        return num_needed_coins[amount]
